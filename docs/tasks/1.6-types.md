# Задача 1.6: Создание типов для программ и уроков

**Статус**: ✅ Выполнено

## Описание

Создать TypeScript интерфейсы для конфигурации уроков, программ и их ивентов. Каждая программа (Browser, Mail, Messenger) расширяется уроком — добавляет свои данные (сайты, письма, чаты) и ивенты (для gameController).

## Типы

### lesson.ts

```ts
interface ComicSlide {
  src: string // URL картинки слайда
}

interface QuickAction {
  id: string
  icon: string // lucide иконка
  label: string // подпись
  onClick: () => void
}

interface LessonConfig {
  id: string
  categoryId: string
  comic?: ComicSlide[] // вступительный комикс (если есть)
  programs: ProgramType[] // какие программы закрепить в taskbar
  quickActions?: QuickAction[] // панель действий слева

  // Расширение программ данными урока
  browserSites?: BrowserSiteConfig[]
  browserBookmarks?: BookmarkConfig[]
  messengerChats?: ChatConfig[]
  mailEmails?: EmailConfig[]
  mailFolders?: MailFolder[]

  // Ивенты для gameController
  events: LessonEvents
}

interface LessonEvents {
  browser?: BrowserEvents
  messenger?: MessengerEvents
  mail?: MailEvents
}
```

### programs.ts

```ts
// --- Browser ---
interface BrowserSiteConfig {
  url: string
  title: string
  component: Component // Vue-компонент страницы
  favicon?: string
}

interface BookmarkConfig {
  label: string
  url: string
  favicon?: string
  icon?: string // lucide иконка (вместо favicon)
}

interface BrowserEvents {
  onUrlNavigate?: (url: string) => void
  onFormSubmit?: (url: string, data: Record<string, string>) => void
  onBookmarkClick?: (url: string) => void
  onTabOpen?: (url: string) => void
  onTabClose?: (url: string) => void
}

// --- Messenger ---
interface ChatConfig {
  id: string
  name: string
  avatar?: string
  messages: MessageConfig[]
  isGroupChat?: boolean
}

interface MessageConfig {
  id: string
  sender: string
  senderAvatar?: string
  text: string
  timestamp?: string
  isOwn?: boolean
}

interface MessengerEvents {
  onMessageSend?: (chatId: string, text: string) => void
  onLinkClick?: (chatId: string, url: string) => void
  onChatOpen?: (chatId: string) => void
}

// --- Mail ---
interface MailFolder {
  id: string
  name: string
  icon?: string
}

interface EmailConfig {
  id: string
  folder: string
  from: string
  fromEmail: string
  to?: string
  subject: string
  body: string // HTML
  date: string
  isRead?: boolean
  attachments?: AttachmentConfig[]
}

interface AttachmentConfig {
  id: string
  name: string
  icon?: string
  isMalicious?: boolean
}

interface MailEvents {
  onMailOpen?: (emailId: string) => void
  onAttachmentOpen?: (emailId: string, attachmentId: string) => void
  onLinkClick?: (emailId: string, url: string) => void
  onReply?: (emailId: string, text: string) => void
}
```
