# Задача 1.19: Создание useUserStore

**Статус**: ⬜ Не выполнено

## Описание

Pinia store для хранения игрового состояния пользователя: HP, прогресс по модулям, статус depleted.
HP — основная игровая валюта: каждая ошибка в уроке снимает HP, при HP = 0 пользователь ждёт восстановления.

> **Не путать с `useUser`** (`app/composables/user/useUser.ts`) — тот стор хранит auth-данные (email, name, isAuthorized). `useUserStore` — **игровой** стор (HP, прогресс уроков).

## Связи

- **useGameController** (1.13) вызывает `hitHp()` при `fail()` и `completeLesson()` при `complete()`.
- **GameOver** (1.14) читает `hp`, `maxHp`, `isDepleted` для отображения оверлея.
- **LessonPage** (1.15) вызывает `syncHp()` при входе и проверяет `isDepleted` перед стартом.

## Реализация стора — `app/stores/useUserStore.ts`

```ts
import type { HpHitResponse, LessonProgress } from '~/lib/api.types'

export const useUserStore = defineStore('userGame', () => {
  // ── State ──
  const hp = ref(3)
  const maxHp = ref(3)
  const restoredAt = ref<string | null>(null)
  const progress = ref<LessonProgress[]>([])
  const currentCategoryId = ref<string | null>(null)

  // ── Getters ──
  const isDepleted = computed(() => hp.value <= 0)

  const hpPercent = computed(() =>
    maxHp.value > 0 ? Math.round((hp.value / maxHp.value) * 100) : 0
  )

  const restoredAtDate = computed(() => (restoredAt.value ? new Date(restoredAt.value) : null))

  const timeUntilRestore = computed(() => {
    if (!restoredAtDate.value) return null
    const diff = restoredAtDate.value.getTime() - Date.now()
    return diff > 0 ? diff : null
  })

  // ── Actions ──

  /**
   * Синхронизировать HP с сервером.
   * Вызывать при входе на страницу урока.
   */
  async function syncHp(categoryId: string) {
    currentCategoryId.value = categoryId
    try {
      // TODO: заменить мок на реальный API
      // const data = await getHp(categoryId)
      const { mockGetHpExtended } = await import('~/mocks/mockHp')
      const data = await mockGetHpExtended(categoryId)
      hp.value = data.current
      maxHp.value = data.max
      restoredAt.value = data.restoredAt
    } catch (e) {
      console.error('[useUserStore] syncHp failed:', e)
    }
  }

  /**
   * Отправить хит HP на сервер.
   * Вызывается из gameController.fail().
   * Возвращает ответ сервера для дальнейшей обработки (GameOver).
   */
  async function hitHp(): Promise<HpHitResponse | null> {
    const catId = currentCategoryId.value
    if (!catId) {
      console.warn('[useUserStore] hitHp: currentCategoryId not set')
      return null
    }
    try {
      // TODO: заменить мок на реальный API
      // const data = await apiHitHp(catId)
      const { mockHitHp } = await import('~/mocks/mockHp')
      const data = await mockHitHp(catId)
      hp.value = data.current
      maxHp.value = data.max
      if (data.isDepleted) {
        restoredAt.value = new Date(Date.now() + 30 * 60_000).toISOString()
      }
      return data
    } catch (e) {
      console.error('[useUserStore] hitHp failed:', e)
      return null
    }
  }

  /**
   * Загрузить прогресс пользователя.
   */
  async function syncProgress() {
    try {
      const { mockGetProgress } = await import('~/mocks/mockProgress')
      const data = await mockGetProgress()
      progress.value = data
    } catch (e) {
      console.error('[useUserStore] syncProgress failed:', e)
    }
  }

  /**
   * Проверить, пройден ли урок.
   */
  function isLessonCompleted(lessonId: string): boolean {
    return progress.value.some((p) => p.lessonId === lessonId)
  }

  /**
   * Добавить урок в прогресс (локально, после POST /lessons/:id/complete).
   */
  function markLessonCompleted(lessonId: string, categoryId: string) {
    if (isLessonCompleted(lessonId)) return
    progress.value.push({
      lessonId,
      categoryId,
      completedAt: new Date().toISOString(),
      attempts: 1
    })
  }

  /**
   * Сброс стора (при логауте или смене пользователя).
   */
  function $reset() {
    hp.value = 3
    maxHp.value = 3
    restoredAt.value = null
    progress.value = []
    currentCategoryId.value = null
  }

  return {
    // state
    hp,
    maxHp,
    restoredAt,
    progress,
    currentCategoryId,
    // getters
    isDepleted,
    hpPercent,
    restoredAtDate,
    timeUntilRestore,
    // actions
    syncHp,
    hitHp,
    syncProgress,
    isLessonCompleted,
    markLessonCompleted,
    $reset
  }
})
```

```ts
onMounted(async () => {
  await userStore.syncHp(categoryId)
  if (userStore.isDepleted) {
    // показать экран «HP закончились»
    return
  }
  // запуск урока…
})
```

### В gameController.fail() (1.13)

```ts
async function fail(reason: string) {
  const userStore = useUserStore()
  const result = await userStore.hitHp()
  triggerGameOver(reason, result)
}
```

### В gameController.complete() (1.13)

```ts
async function complete() {
  const userStore = useUserStore()
  await completeLesson(config.id)
  userStore.markLessonCompleted(config.id, config.categoryId)
}
```
